import { TypeTable } from 'fumadocs-ui/components/type-table';

## AlbumService

```ts
import { AlbumService } from 'jiosaavn-sdk'

const albumService = new AlbumService()
```

---

### getAlbumById

Fetch an album by its ID.

```ts
const album = await albumService.getAlbumById('23241654')
```

#### Parameters

<TypeTable
  type={{
    albumId: {
      description: 'The unique album ID',
      type: 'string',
      required: true,
    },
  }}
/>

**Returns:** `Promise<AlbumModel>`

**Throws:** `SaavnError(404)` if the album is not found.

---

### getAlbumByLink

Fetch an album by a JioSaavn link token.

```ts
const album = await albumService.getAlbumByLink('ITIyo-GDr7A_')
```

#### Parameters

<TypeTable
  type={{
    albumLink: {
      description: 'The token extracted from a JioSaavn album URL',
      type: 'string',
      required: true,
    },
  }}
/>

**Returns:** `Promise<AlbumModel>`

**Throws:** `SaavnError(404)` if the album is not found.
