import { TypeTable } from 'fumadocs-ui/components/type-table';

## SearchService

```ts
import { SearchService } from 'jiosaavn-sdk'

const searchService = new SearchService()
```

---

### searchAll

Search across all content types at once.

```ts
const results = await searchService.searchAll('Dua Lipa')
```

#### Parameters

<TypeTable
  type={{
    query: {
      description: 'The search query',
      type: 'string',
      required: true,
    },
  }}
/>

**Returns:** `Promise<SearchModel>`

**Throws:** `SaavnError(404)` if no results are found.

---

### searchSongs

Search specifically for songs.

```ts
const results = await searchService.searchSongs({
  query: 'Levitating',
  page: 0,
  limit: 10
})
```

#### Parameters

<TypeTable
  type={{
    query: {
      description: 'The search query',
      type: 'string',
      required: true,
    },
    page: {
      description: 'Page number',
      type: 'number',
      required: true,
    },
    limit: {
      description: 'Max results per page',
      type: 'number',
      required: true,
    },
  }}
/>

#### Return Type — SearchSongModel

<TypeTable
  type={{
    total: {
      description: 'Total number of matches',
      type: 'number',
      required: true,
    },
    start: {
      description: 'Pagination offset',
      type: 'number',
      required: true,
    },
    results: {
      description: 'Array of song objects',
      type: 'SongModel[]',
      required: true,
    },
  }}
/>

## SearchModel

Returned by `searchAll`. Contains categorized results, each with a `results` array and `position` ranking.

<TypeTable
  type={{
    topQuery: {
      description: 'Top matching results across all types',
      type: 'SearchResponseModel<SearchTopQueryItem[]>',
      required: true,
    },
    songs: {
      description: 'Matching songs',
      type: 'SearchResponseModel<SearchSongItem[]>',
      required: true,
    },
    albums: {
      description: 'Matching albums',
      type: 'SearchResponseModel<SearchAlbumItem[]>',
      required: true,
    },
    artists: {
      description: 'Matching artists',
      type: 'SearchResponseModel<SearchArtistItem[]>',
      required: true,
    },
    playlists: {
      description: 'Matching playlists',
      type: 'SearchResponseModel<SearchPlaylistItem[]>',
      required: true,
    },
  }}
/>
